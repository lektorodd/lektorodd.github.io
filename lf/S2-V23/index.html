    A_kar = sum(rng.normal(3.8, 1.2, A))
    B_kar = sum(rng.normal(3.4, 1.4, B))
    C_kar = sum(rng.normal(4.1, 1.1, C))

    snitt = (A_kar + B_kar + C_kar) / N

    return snitt
