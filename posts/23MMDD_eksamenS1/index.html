import matplotlib.pyplot as plt
import numpy as np

år = [0, 20, 30, 40, 51]
gårdar = [154977, 99382, 68539, 47688, 38633]

plt.plot(år, gårdar, "ro")
plt.show()

from scipy.optimize import curve_fit

def f(x, a, b):
    return a*b**x

K, E = curve_fit(f, år, gårdar)

a, b = K
